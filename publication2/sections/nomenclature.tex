\section*{Nomenclature}
\label{sec:nomenclature}

\noindent Units: length \si{mm}, time \si{s}, temperature \si{K}, stress \si{MPa}.

\begin{description}[leftmargin=2.6cm,style=nextline]
  \item[$T$] Temperature.
  \item[$\rho$] Density.
  \item[$c_p$] Specific heat.
  \item[$k$] Thermal conductivity.
  \item[$h_{\mathrm{conv}}$] Convective heat-transfer coefficient.
  \item[$\vect{u}$] Displacement vector.
  \item[$\tensor{D}$] Elastic stiffness tensor.
  \item[$\alpha$] Coefficient of thermal expansion.
  \item[$\boldsymbol{\varepsilon}^{\mathrm{th}}$] Thermal strain.
  \item[$\boldsymbol{\varepsilon}^{\mathrm{inh}}$] Inherent-strain surrogate.
  \item[$E$] Line energy (energy per unit length).
  \item[$v$] Heat-source travel speed.
  \item[$r_0$] Gaussian heat-source radius.
  \item[$q_0$] Peak surface heat flux.
  \item[$N$] Number of passes.
\end{description}
